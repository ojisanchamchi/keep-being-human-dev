class BitSet
  def initialize(size)
    @size = size
    @bits = Array.new((size / 32.0).ceil, 0)
  end
  
  def [](index)

    word_index = index / 32
    bit_index = index % 32

  def []=(index, value)

    word_index = index / 32
    bit_index = index % 32

      @bits[word_index] |= (1 << bit_index)

      @bits[word_index] &= ~(1 << bit_index)

  end

  end

  end

module EnumeratorChaining
  refine Enumerator do
    def chain(other)

    def zip_with(other)
      Enumerator.new do |yielder|
        loop do
          yielder << [self.next, other.next]
        end
      rescue StopIteration
        nil
      end
    end
    
    def interleave(other)
      Enumerator.new do |yielder|
        loop do
          yielder << self.next
          yielder << other.next

    end

        buffer = []
        
        each do |item|
          buffer << item

          end
        end

      end
    end
  end
end
